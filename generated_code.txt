def is_prime(number):
  """
  Returns True if the given number is prime.

  Args:
    number: The number to check.

  Returns:
    True if the number is prime, False otherwise.
  """

  if number == 2:
    return True
  elif number <= 1 or number % 2 == 0:
    return False

  for i in range(3, int(number ** 0.5) + 1, 2):
    if number % i == 0:
      return False

  return True
