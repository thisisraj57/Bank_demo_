def is_prime(num):
    if num <= 1:
        return False  # 1 and negative numbers are not prime
    for i in range(2, num):
        if num % i == 0:
            return False  # If any number between 2 and num-1 divides num evenly, it's not prime
    return True  # If no number between 2 and num-1 divides num evenly, it's prime

# Example usage
number = int(input("Enter a number: "))
if is_prime(number):
    print(f"{number} is a prime number.")
else:
    print(f"{number} is not a prime number.")
